        )}

        {/* Modal for Reservation Preview */}
        {showReservationPreviewModal && previewReservation && (
          <div 
            className="fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center z-50 p-4"
            onClick={(e) => {
              if (e.target === e.currentTarget) {
                setShowReservationPreviewModal(false)
                setPreviewReservation(null)
              }
            }}
          >
            <div className="bg-white rounded-lg p-6 w-full max-w-md">
              <div className="flex justify-between items-center mb-4">
                <h3 className="text-lg font-semibold">Предпросмотр бронирования</h3>
                <button
                  onClick={() => {
                    setShowReservationPreviewModal(false)
                    setPreviewReservation(null)
                  }}
                  className="text-gray-400 hover:text-gray-600 transition-colors"
                >
                  ✕
                </button>
              </div>
              
              <div className="space-y-4">
                <div>
                  <label className="block text-sm font-medium text-gray-700">Номер бронирования</label>
                  <p className="text-sm text-gray-900">{previewReservation.reservationNumber}</p>
                </div>
                
                <div>
                  <label className="block text-sm font-medium text-gray-700">Пользователь</label>
                  <p className="text-sm text-gray-900">{previewReservation.userName}</p>
                </div>
                
                <div>
                  <label className="block text-sm font-medium text-gray-700">Ресторан</label>
                  <p className="text-sm text-gray-900">{previewReservation.restaurantName}</p>
                </div>
                
                <div>
                  <label className="block text-sm font-medium text-gray-700">Дата и время</label>
                  <p className="text-sm text-gray-900">{previewReservation.date} в {previewReservation.time}</p>
                </div>
                
                <div>
                  <label className="block text-sm font-medium text-gray-700">Количество гостей</label>
                  <p className="text-sm text-gray-900">{previewReservation.guests}</p>
                </div>
                
                <div>
                  <label className="block text-sm font-medium text-gray-700">Статус</label>
                  <span className={`px-2 py-1 rounded-full text-xs font-medium ${
                    previewReservation.status === 'confirmed' ? 'bg-green-100 text-green-800' :
                    previewReservation.status === 'pending' ? 'bg-yellow-100 text-yellow-800' :
                    'bg-red-100 text-red-800'
                  }`}>
                    {previewReservation.status === 'confirmed' ? 'Подтверждено' :
                     previewReservation.status === 'pending' ? 'Ожидает' : 'Отменено'}
                  </span>
                </div>
                
                {previewReservation.specialRequests && (
                  <div>
                    <label className="block text-sm font-medium text-gray-700">Особые пожелания</label>
                    <p className="text-sm text-gray-900">{previewReservation.specialRequests}</p>
                  </div>
                )}
              </div>
              
              <div className="flex space-x-3 mt-6">
                {previewReservation.status === 'pending' && (
                  <button
                    onClick={() => handleUpdateReservationStatus(previewReservation._id, 'confirmed')}
                    className="flex-1 bg-green-500 text-white px-4 py-2 rounded-md hover:bg-green-600 transition-colors"
                  >
                    Подтвердить
                  </button>
                )}
                
                {previewReservation.status === 'confirmed' && (
                  <button
                    onClick={() => handleUpdateReservationStatus(previewReservation._id, 'completed')}
                    className="flex-1 bg-blue-500 text-white px-4 py-2 rounded-md hover:bg-blue-600 transition-colors"
                  >
                    Завершить
                  </button>
                )}
                
                {previewReservation.status !== 'cancelled' && (
                  <button
                    onClick={() => handleUpdateReservationStatus(previewReservation._id, 'cancelled')}
                    className="flex-1 bg-red-500 text-white px-4 py-2 rounded-md hover:bg-red-600 transition-colors"
                  >
                    Отменить
                  </button>
                )}
              </div>
            </div>
          </div>
        )}
